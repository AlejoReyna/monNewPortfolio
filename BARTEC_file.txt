INSERT INTO #TMPSALIDASPEIXDIA
select  STPRECFOLIO         AS FOLIO
       ,1					AS TIPO
	   ,FEC_TX				AS FECTX
	   ,NOMBREORDENANTE     AS NOMBRE
	   ,CLAVERASTREO        AS CLAVERASTREO
	   ,CUENTAORDENANTE     AS CUENTA
	   ,INSTITUCIONORDENANTE AS BANCO
	   ,RFCCURPORDENANTE    AS RFCCURP
	   ,CONCEPTOPAGO        AS CONCEPTO
	   ,MENSAJE             AS RESULTADO
	   ,CASE WHEN RESPCODE='00' THEN 1 ELSE 0 END AS ESTATUS
	   ,MOV_STPSPEIREC.MONTO AS MONTO
	   ,MOV_STPSPEIREC.IMP_COMISION AS COMISION
from	InveraterDBC..mov_stpspeirec
where	CTEID   		= @ID_CTECORP
  and	fec_tx			between @FecIni and @FecFin

INSERT INTO #TMPSALIDASPEIXDIA
select  STPENVFOLIO         AS FOLIO
       ,1					AS TIPO
	   ,FEC_TX				AS FECTX
	   ,NOMBREORDENANTE     AS NOMBRE
	   ,CLAVERASTREO        AS CLAVERASTREO
	   ,CUENTAORDENANTE     AS CUENTA
	   ,INSTITUCIONORDENANTE AS BANCO
	   ,RFCCURPORDENANTE    AS RFCCURP
	   ,CONCEPTOPAGO        AS CONCEPTO
	   ,TXTSTPSENDCOMMENTS             AS RESULTADO
	   ,CASE WHEN ID_NUM_STPENVSTAT='13' THEN 1 ELSE 0 END AS ESTATUS

	   ,MOV_STPSPEIENV.IMP_COMISION AS COMISION

INSERT INTO #TMPTOTALES
SELECT  TIPO 
       ,COUNT(*)
	   ,SUM(MONTO) 
	   ,MAX(FOLIO)
FROM   #TMPSALIDASPEIXDIA
GROUP BY TIPO
	SELECT TIPO 
       ,SUM(CANTOTAL)	 AS CANTOTAL
	   ,SUM(IMPTOTAL)	 AS IMPTOTAL
	   ,MAX(ULTIMOFOLIO) AS ULTIMOFOLIO
	FROM   #TMPTOTALES
	GROUP BY TIPO
	ORDER BY TIPO

	SELECT ID_NUM_INSTFIN
      ,NUM_REFSTP
	  ,RIGHT(CONVERT(VARCHAR,NUM_REFSTP),3) + ' ' + DESC_INSTFIN AS DESC_INSTFIN
	  ,ABREV_INSTFIN
	FROM  InveraterDBC..INSTFIN
	WHERE BIT_PERMITESPEI = 1
	ORDER BY DESC_INSTFIN
